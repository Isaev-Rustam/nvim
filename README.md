local discipline = require("craftzdog.discipline")

discipline.cowboy()

local keymap = vim.keymap
local opts = { noremap = true, silent = true }

-- –í—ã–ø–æ–ª–Ω–µ–Ω–∏–µ –¥–µ–π—Å—Ç–≤–∏–π –±–µ–∑ –∏–∑–º–µ–Ω–µ–Ω–∏—è —Ä–µ–≥–∏—Å—Ç—Ä–æ–≤
keymap.set("n", "x", '"_x') -- –£–¥–∞–ª–∏—Ç—å —Å–∏–º–≤–æ–ª –±–µ–∑ –∫–æ–ø–∏—Ä–æ–≤–∞–Ω–∏—è –≤ —Ä–µ–≥–∏—Å—Ç—Ä
keymap.set("n", "<Leader>p", '"0p') -- –í—Å—Ç–∞–≤–∏—Ç—å –∏–∑ —Ä–µ–≥–∏—Å—Ç—Ä–∞ 0
keymap.set("n", "<Leader>P", '"0P') -- –í—Å—Ç–∞–≤–∏—Ç—å –∏–∑ —Ä–µ–≥–∏—Å—Ç—Ä–∞ 0 –ø–µ—Ä–µ–¥ –∫—É—Ä—Å–æ—Ä–æ–º
keymap.set("v", "<Leader>p", '"0p') -- –í—Å—Ç–∞–≤–∏—Ç—å –∏–∑ —Ä–µ–≥–∏—Å—Ç—Ä–∞ 0 –≤ –≤–∏–∑—É–∞–ª—å–Ω–æ–º —Ä–µ–∂–∏–º–µ
keymap.set("n", "<Leader>c", '"_c') -- –ó–∞–º–µ–Ω–∏—Ç—å —Ç–µ–∫—Å—Ç –±–µ–∑ –∫–æ–ø–∏—Ä–æ–≤–∞–Ω–∏—è –≤ —Ä–µ–≥–∏—Å—Ç—Ä
keymap.set("n", "<Leader>C", '"_C') -- –ó–∞–º–µ–Ω–∏—Ç—å –¥–æ –∫–æ–Ω—Ü–∞ —Å—Ç—Ä–æ–∫–∏ –±–µ–∑ –∫–æ–ø–∏—Ä–æ–≤–∞–Ω–∏—è
keymap.set("v", "<Leader>c", '"_c') -- –ó–∞–º–µ–Ω–∏—Ç—å –≤—ã–¥–µ–ª–µ–Ω–Ω—ã–π —Ç–µ–∫—Å—Ç –±–µ–∑ –∫–æ–ø–∏—Ä–æ–≤–∞–Ω–∏—è
keymap.set("v", "<Leader>C", '"_C') -- –ó–∞–º–µ–Ω–∏—Ç—å –≤—ã–¥–µ–ª–µ–Ω–Ω—ã–π —Ç–µ–∫—Å—Ç –¥–æ –∫–æ–Ω—Ü–∞ —Å—Ç—Ä–æ–∫–∏
keymap.set("n", "<Leader>d", '"_d') -- –£–¥–∞–ª–∏—Ç—å —Ç–µ–∫—Å—Ç –±–µ–∑ –∫–æ–ø–∏—Ä–æ–≤–∞–Ω–∏—è –≤ —Ä–µ–≥–∏—Å—Ç—Ä
keymap.set("n", "<Leader>D", '"_D') -- –£–¥–∞–ª–∏—Ç—å –¥–æ –∫–æ–Ω—Ü–∞ —Å—Ç—Ä–æ–∫–∏ –±–µ–∑ –∫–æ–ø–∏—Ä–æ–≤–∞–Ω–∏—è
keymap.set("v", "<Leader>d", '"_d') -- –£–¥–∞–ª–∏—Ç—å –≤—ã–¥–µ–ª–µ–Ω–Ω—ã–π —Ç–µ–∫—Å—Ç –±–µ–∑ –∫–æ–ø–∏—Ä–æ–≤–∞–Ω–∏—è
keymap.set("v", "<Leader>D", '"_D') -- –£–¥–∞–ª–∏—Ç—å –≤—ã–¥–µ–ª–µ–Ω–Ω—ã–π —Ç–µ–∫—Å—Ç –¥–æ –∫–æ–Ω—Ü–∞ —Å—Ç—Ä–æ–∫–∏

-- –£–≤–µ–ª–∏—á–µ–Ω–∏–µ/—É–º–µ–Ω—å—à–µ–Ω–∏–µ —á–∏—Å–ª–∞
keymap.set("n", "+", "<C-a>") -- –£–≤–µ–ª–∏—á–∏—Ç—å —á–∏—Å–ª–æ
keymap.set("n", "-", "<C-x>") -- –£–º–µ–Ω—å—à–∏—Ç—å —á–∏—Å–ª–æ

-- –£–¥–∞–ª–µ–Ω–∏–µ —Å–ª–æ–≤–∞ –Ω–∞–∑–∞–¥
keymap.set("n", "dw", 'vb"_d') -- –£–¥–∞–ª–∏—Ç—å —Å–ª–æ–≤–æ –Ω–∞–∑–∞–¥ –±–µ–∑ –∫–æ–ø–∏—Ä–æ–≤–∞–Ω–∏—è

-- –í—ã–¥–µ–ª–∏—Ç—å –≤—Å—ë
keymap.set("n", "<C-a>", "gg<S-v>G") -- –í—ã–¥–µ–ª–∏—Ç—å –≤–µ—Å—å —Ç–µ–∫—Å—Ç

-- –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ —Å –ø—Ä–∞–≤–∞–º–∏ root (–ø–æ–∫–∞ –Ω–µ —Ä–∞–±–æ—Ç–∞–µ—Ç)
-- vim.api.nvim_create_user_command('W', 'w !sudo tee > /dev/null %', {})

-- –û—Ç–∫–ª—é—á–µ–Ω–∏–µ –ø—Ä–æ–¥–æ–ª–∂–µ–Ω–∏–π
keymap.set("n", "<Leader>o", "o<Esc>^Da", opts) -- –í—Å—Ç–∞–≤–∏—Ç—å –Ω–æ–≤—É—é —Å—Ç—Ä–æ–∫—É –Ω–∏–∂–µ
keymap.set("n", "<Leader>O", "O<Esc>^Da", opts) -- –í—Å—Ç–∞–≤–∏—Ç—å –Ω–æ–≤—É—é —Å—Ç—Ä–æ–∫—É –≤—ã—à–µ

-- –ü–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ –ø–æ —Å–ø–∏—Å–∫—É –ø–µ—Ä–µ—Ö–æ–¥–æ–≤
keymap.set("n", "<C-m>", "<C-i>", opts) -- –ü–µ—Ä–µ–π—Ç–∏ –≤–ø–µ—Ä—ë–¥ –ø–æ —Å–ø–∏—Å–∫—É –ø–µ—Ä–µ—Ö–æ–¥–æ–≤

-- –†–∞–±–æ—Ç–∞ —Å –≤–∫–ª–∞–¥–∫–∞–º–∏
keymap.set("n", "te", ":tabedit") -- –û—Ç–∫—Ä—ã—Ç—å –Ω–æ–≤—É—é –≤–∫–ª–∞–¥–∫—É
keymap.set("n", "<tab>", ":tabnext<Return>", opts) -- –ü–µ—Ä–µ–π—Ç–∏ –∫ —Å–ª–µ–¥—É—é—â–µ–π –≤–∫–ª–∞–¥–∫–µ
keymap.set("n", "<s-tab>", ":tabprev<Return>", opts) -- –ü–µ—Ä–µ–π—Ç–∏ –∫ –ø—Ä–µ–¥—ã–¥—É—â–µ–π –≤–∫–ª–∞–¥–∫–µ

-- –†–∞–∑–¥–µ–ª–µ–Ω–∏–µ –æ–∫–Ω–∞
keymap.set("n", "ss", ":split<Return>", opts) -- –ì–æ—Ä–∏–∑–æ–Ω—Ç–∞–ª—å–Ω–æ–µ —Ä–∞–∑–¥–µ–ª–µ–Ω–∏–µ
keymap.set("n", "sv", ":vsplit<Return>", opts) -- –í–µ—Ä—Ç–∏–∫–∞–ª—å–Ω–æ–µ —Ä–∞–∑–¥–µ–ª–µ–Ω–∏–µ

-- –ü–µ—Ä–µ–º–µ—â–µ–Ω–∏–µ –º–µ–∂–¥—É –æ–∫–Ω–∞–º–∏
keymap.set("n", "sh", "<C-w>h") -- –í–ª–µ–≤–æ
keymap.set("n", "sk", "<C-w>k") -- –í–≤–µ—Ä—Ö
keymap.set("n", "sj", "<C-w>j") -- –í–Ω–∏–∑
keymap.set("n", "sl", "<C-w>l") -- –í–ø—Ä–∞–≤–æ

-- –ò–∑–º–µ–Ω–µ–Ω–∏–µ —Ä–∞–∑–º–µ—Ä–∞ –æ–∫–Ω–∞
keymap.set("n", "<C-w><left>", "<C-w><") -- –£–º–µ–Ω—å—à–∏—Ç—å —à–∏—Ä–∏–Ω—É
keymap.set("n", "<C-w><right>", "<C-w>>") -- –£–≤–µ–ª–∏—á–∏—Ç—å —à–∏—Ä–∏–Ω—É
keymap.set("n", "<C-w><up>", "<C-w>+") -- –£–≤–µ–ª–∏—á–∏—Ç—å –≤—ã—Å–æ—Ç—É
keymap.set("n", "<C-w><down>", "<C-w>-") -- –£–º–µ–Ω—å—à–∏—Ç—å –≤—ã—Å–æ—Ç—É

-- –î–∏–∞–≥–Ω–æ—Å—Ç–∏–∫–∞
keymap.set("n", "<C-j>", function()
    vim.diagnostic.goto_next() -- –ü–µ—Ä–µ–π—Ç–∏ –∫ —Å–ª–µ–¥—É—é—â–µ–π –¥–∏–∞–≥–Ω–æ—Å—Ç–∏–∫–µ
end, opts)

-- –ó–∞–º–µ–Ω–∞ HEX –Ω–∞ HSL
keymap.set("n", "<leader>r", function()
    require("craftzdog.hsl").replaceHexWithHSL()
end)

-- –ü–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏–µ –ø–æ–¥—Å–∫–∞–∑–æ–∫ LSP
keymap.set("n", "<leader>i", function()
    require("craftzdog.lsp").toggleInlayHints()
end)

-- –ö–æ–º–∞–Ω–¥–∞ –¥–ª—è –ø–µ—Ä–µ–∫–ª—é—á–µ–Ω–∏—è –∞–≤—Ç–æ—Ñ–æ—Ä–º–∞—Ç–∏—Ä–æ–≤–∞–Ω–∏—è

end, {})
